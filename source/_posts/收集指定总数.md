## 源代码

``` swift
let totalGems = randomNumberOfGems
var gemCounter = 0
while gemCounter < totalGems {
    if !isBlocked {
        moveForward()
    } else if !isBlockedRight {
        turnRight()
    } else if !isBlockedLeft {
        turnLeft()
    } else {
        turnLeft()
        turnLeft()
    }
    if isOnGem {
        collectGem()
        gemCounter += 1
    }
}
```
