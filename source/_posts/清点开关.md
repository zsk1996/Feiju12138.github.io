## 源代码

``` swift
var gemCounter = 0
var switchCounter = 0
moveForward()
for i in 1 ... 11 {
    if !isBlocked {
        moveForward()
    } else if !isBlockedRight {
        turnRight()
    }
    if isOnGem {
        collectGem()
        gemCounter += 1
    }
}
for i in 1 ... 11 {
    if !isBlocked {
        moveForward()
    } else if !isBlockedRight {
        turnRight()
    }
    if isOnClosedSwitch && switchCounter != gemCounter {
        toggleSwitch()
        switchCounter += 1
    }
}
```
