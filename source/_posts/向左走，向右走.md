## 源代码

``` swift
func navigateAroundWall() {
    moveForward()
    if isBlocked {
        if !isBlockedLeft {
            turnLeft()
        } else if !isBlockedRight {
            turnRight()
        }
    }
}

while true {
    navigateAroundWall()
    if isOnGem {
        collectGem()
    }
    if isOnClosedSwitch {
        toggleSwitch()
    }
    if isBlockedRight && isBlocked && isBlockedLeft {
        break
    }
}
```
