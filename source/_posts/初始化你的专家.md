## 源代码

``` swift
let expert = Expert()
for i in 1 ... 4 {
    expert.moveForward()
    expert.moveForward()
    expert.moveForward()
    if i == 1 {
        expert.turnLockUp()
    } else {
        expert.collectGem()
    }
    expert.turnLeft()
    expert.turnLeft()
    expert.moveForward()
    expert.moveForward()
    expert.moveForward()
    expert.turnLeft()
}
```
