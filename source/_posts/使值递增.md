## 源代码

``` swift
var gemCounter = 0
while !isBlocked || !isBlockedLeft || !isBlockedRight {
    if isBlocked && isBlockedLeft {
        turnRight()
        moveForward()
    } else {
        moveForward()
    }
    if isOnGem {
        collectGem()
        gemCounter += 1
    }
}
```
